In `app/ui/screens/checkin_screen.py`, locate `_on_checkin` and `_on_checkout` methods. Refactor (or add) so they call a shared `_refresh_after_action` helper:

```python
    def _on_checkin(self, period_name: str) -> None:
        """签到回调 — 调 service + 刷新所有 PeriodCard。"""
        if not self._checkin_service:
            return
        try:
            self._checkin_service.checkin(self._date_str, period_name)
        except Exception as e:
            Logger.error(f"CheckinScreen: 签到失败 {e}")
            return
        self._refresh_after_action()

    def _on_checkout(self, period_name: str) -> None:
        """签退回调 — 调 service + 刷新所有 PeriodCard。"""
        if not self._checkin_service:
            return
        try:
            self._checkin_service.checkout(self._date_str, period_name)
        except Exception as e:
            Logger.error(f"CheckinScreen: 签退失败 {e}")
            return
        self._refresh_after_action()

    def _refresh_after_action(self) -> None:
        """从 service 拉最新 day_status, push 给所有 PeriodCard + StatusBox。"""
        day_status = self._checkin_service.get_day_status(self._date_str)
        if day_status is None:
            return
        for period_name, card in self._period_cards.items():
            ps = next((p for p in day_status.periods if p.period == period_name), None)
            card.set_status_from_period(ps)
        if self._status_box is not None:
```
